function getCount(str) {
    let vowels = ["a", "e", "i", "o", "u"];
    let count = 0;
    for (i = 0; i < str.length; i++){
        for (j = 0; j < vowels.length; j++){
            if(str[i] === vowels[j]) {
                count += 1;
            }
        }
    }
    return count;
}
